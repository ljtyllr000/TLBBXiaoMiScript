function 挂内息()
   屏幕提示("即将开内息丹和经验丹刷内息, 刷完自动返回大理") 
   屏幕提示("战为不负卿！")

   -- 使用内息丹
   if 获取背包物品数量("善武丸") > 0 then
      右键使用物品("善武丸",1)     
   end
   if 获取背包物品数量("从武丸") > 0 then
      右键使用物品("从武丸",1)
   end

   -- 使用经验丹
   if 获取背包物品数量("醒灵玉") > 0 then
      右键使用物品("醒灵玉",1)
   end
   if 获取背包物品数量("聚灵玉") > 0 then
      右键使用物品("聚灵玉",1)
   end

   执行功能("挂机打怪")
   组队_离开队伍()
   执行功能("大理回城")

end


挂内息()
